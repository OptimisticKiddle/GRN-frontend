4.1 Loading data

  • The diff footprint csv file can be downloaded from the download page
xxxxxxxxxx
pb_gene = "SMARCA4"  # The perturbed gene
diff_footprint = read.csv("./id-3_SMARCA4_BT869_diff_footprint.csv",head=TRUE)
diff_tf = diff_footprint["tf"]
diff_tf = rbind(diff_tf, pb_gene)
# String processing for subsequent TF matches
diff_tf[,"tf"] <- str_replace_all(diff_tf[,"tf"],"\\((.*?)\\)","")
diff_tf_adjust = diff_tf
n_rows = nrow(diff_tf)
for(i in n_rows:1) {
    split_strs_1 = strsplit(diff_tf[i,"tf"], '::')
    if (length(split_strs_1[[1]])>=2) {
        diff_tf_adjust = as.data.frame(diff_tf_adjust[-i,])
        for(j in 1:length(split_strs_1[[1]])){
            diff_tf_adjust = rbind(diff_tf_adjust, split_strs_1[[1]][j])
        }
    }
}
diff_tf_adjust = data.frame(diff_tf_adjust[!duplicated(diff_tf_adjust), ])
colnames(diff_tf_adjust) <- c("tf")
diff_tf = diff_tf_adjust

4.3 Regulatory networks DIY

  • Draw an interaction network diagram of footprint related proteins (DIY version).
xxxxxxxxxx
data_links <- data_mapped$STRING_id[1:100] %>% string_db$get_interactions()
links <- data_links %>%
  mutate(from = data_mapped[match(from, data_mapped$STRING_id), "tf"]) %>% 
  mutate(to = data_mapped[match(to, data_mapped$STRING_id), "tf"]) %>%  
  dplyr::select(from, to , last_col()) %>% 
  dplyr::rename(weight = combined_score)
links <- links %>% mutate(from_c = count(., from)$n[match(from, count(., from)$from)]) %>%
  mutate(to_c = count(., to)$n[match(to, count(., to)$to)]) %>%
  filter(!(from_c == 1 & to_c == 1)) %>%
  dplyr::select(1,2,3)
nodes <- links %>% { data.frame(gene = c(.$from, .$to)) } %>% distinct()
net <- igraph::graph_from_data_frame(d=links,vertices=nodes,directed = F)
igraph::V(net)$deg <- igraph::degree(net)
igraph::V(net)$size <- igraph::degree(net)/5
igraph::E(net)$width <- igraph::E(net)$weight/10
ggraph(net,layout = "kk")+
  geom_edge_fan(aes(edge_width=width), color = "lightblue", show.legend = F)+
  geom_node_point(aes(size=size), color="orange", alpha=0.7)+
  geom_node_text(aes(filter=deg>5, label=name),repel = T)+
  scale_edge_width(range = c(0.2,1))+
  scale_size_continuous(range = c(1,10) )+ 
  theme_graph() + ggtitle("The perturbed gene: SMARCA4") + theme(plot.title = element_text(hjust = 0.5))
